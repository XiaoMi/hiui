import React from 'react'
import Switch from '@hi-ui/hiui/es/switch'
import Button from '@hi-ui/hiui/es/button'
class Demo extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      checked: true,
      disabled: true
    }
  }
  onChange (status) {
    console.log(status)
    this.setState({checked: status})
  }
  render () {
    return (
      <div>
        <p>
          <Button onClick={() => {this.setState({disabled: !this.state.disabled})}}></Button>
          <Button onClick={() => {this.setState({checked: !this.state.checked})}}></Button>
        </p >
        <Switch checked={this.state.checked} disabled={this.state.disabled} content={['', '']} onChange={this.onChange.bind(this)}/>
      </div>
    )
  }
}
