Alert 警告提示

作用于页面的内容区域的提示，非触发类信息

何时使用

警示，内容提示，说明，要求，规则等阅读类信息

显示在页面局部的提示性文案

import React from 'react'
import Alert from '@hi-ui/hiui/es/alert'
class Demo extends React.Component {
  render () {
    return (
      <div>
        <Alert type="primary" title="信息提示的文案" onClose={()=>{console.log('alert关闭回调')}} />
        <br />
        <Alert type="success" title="成功提示的文案" onClose={()=>{console.log('alert关闭回调')}} />

      </div>
    )
  }
}
