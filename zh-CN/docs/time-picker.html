TimePicker 时间选择器

基础

import React from 'react'
import TimePicker from '@hi-ui/hiui/es/time-picker/TimePicker'
class Demo extends React.Component {
  render() {
    return (
      <TimePicker
        value={new Date()}
        format="HH:mm:ss"
        onChange={(date, dateString) => console.log('时间', date, dateString)}
      />
    )
  }
}

格式

import React from 'react'
import TimePicker from '@hi-ui/hiui/es/time-picker/TimePicker'
class Demo extends React.Component {
  render() {
    return (
      <div>
        <TimePicker
          value={new Date()}
          format="HH:mm:ss"
          onChange={(date, dateString) => console.log(date, dateString)}
        />
        <TimePicker
          value={new Date()}
          format="HH:mm"
          onChange={(date, dateString) => console.log(date, dateString)}
        />
        <TimePicker
          value={new Date()}
          format="HH"
          onChange={(date, dateString) => console.log(date, dateString)}
        />
      </div>
    )
  }
}

时间范围

至
import React from 'react'
import TimePicker from '@hi-ui/hiui/es/time-picker/TimePicker'
class Demo extends React.Component {
  render() {
    return (
      <TimePicker
        value={{start: new Date(), end: new Date()}}
        type="timerange"
        format="HH:mm:ss"
        onChange={date => console.log('时间范围', date)}
      />
    )
  }
}

禁选时间

import React from 'react'
import TimePicker from '@hi-ui/hiui/es/time-picker/TimePicker'
class Demo extends React.Component {
  render() {
    return (
      <TimePicker
        value={new Date()}
        format="HH:mm:ss"
        disabledHours={[2, 3, 4, 5, 6]}
        disabledMinutes={(selectedHour) => {
          if (selectedHour > 12) {
            const arr = []
            for (let i=0; i<12; i++) {
              arr.push(i)
            }
            return arr
          }
        }}
        disabledSeconds={(selectedHour, selectedMinute) => {
          if (selectedHour > 12 && selectedMinute > 12) {
            return [30, 31, 32, 33, 34, 35]
          }
          return [0, 1, 2, 3]
        }}
        onChange={(date, dateString) => console.log(date, dateString)}
      />
    )
  }
}
